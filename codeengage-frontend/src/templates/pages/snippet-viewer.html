1import { useState, useEffect, useCallback } from 'react';
2
3/**
4 * Custom hook for managing API data with loading and error states
5 * @param {string} url - API endpoint to fetch data from
6 * @param {Object} options - Fetch options
7 * @returns {Object} - Data, loading state, error state, and refetch function
8 */
9const useApiData = (url, options = {}) => {
10  const [data, setData] = useState(null);
11  const [loading, setLoading] = useState(true);
12  const [error, setError] = useState(null);
13
14  const fetchData = useCallback(async () => {
15    try {
16      setLoading(true);
17      setError(null);
18
19      const response = await fetch(url, {
20        method: 'GET',
21        headers: {
22          'Content-Type': 'application/json',
23          ...options.headers,
24        },
25        ...options,
26      });
27
28      if (!response.ok) {
29        throw new Error(`HTTP error! status: ${response.status}`);
30      }
31
32      const result = await response.json();
33      setData(result);
34    } catch (err) {
35      setError(err.message);
36    } finally {
37      setLoading(false);
38    }
39  }, [url, options]);
40
41  useEffect(() => {
42    fetchData();
43  }, [fetchData]);
44
45  return { data, loading, error, refetch: fetchData };
46};
47
48export default useApiData;
